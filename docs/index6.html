# Match URLs in a File

import re

def extract_urls(filename):
    with open(filename, 'r') as file:
        content = file.read()
        urls = re.findall(r'https?://[^\s]+', content)
        return urls

# Example usage
filename = 'sample.txt'

urls = extract_urls(filename)
print("URLs found:", urls)

URLs found: ['https://www.example.com']

#  Match All Numbers in a File

import re

def extract_numbers(filename):
    with open(filename, 'r') as file:
        content = file.read()
        numbers = re.findall(r'\b\d+(\.\d+)?\b', content)
        return numbers

# Example usage
filename = 'sample.txt'

numbers = extract_numbers(filename)

# Get Nested Attributes in a Deeply Nested Object

from magicattr import get

class Inner:
    def __init__(self, value):
        self.value = value

class Outer:
    def __init__(self, inner):
        self.inner = inner

inner = Inner("Hello")

outer = Outer(inner)

# Access deeply nested attribute
value = get(outer, "inner.value")

value

'Hello'

# Find and Remove Extra Whitespaces

import re

def remove_extra_spaces(filename):
    with open(filename, 'r') as file:
        content = file.read()
        updated_content = re.sub(r'\s+', ' ', content).strip()

    with open(filename, 'w') as file:
        file.write(updated_content)

    return "Extra spaces removed."

# Example usage
filename = 'sample.txt'
print(remove_extra_spaces(filename))
