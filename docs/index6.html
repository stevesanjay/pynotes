# Extract All Hexadecimal Values

import re

def extract_hex_values(filename):
    with open(filename, 'r') as file:
        content = file.read()
        hex_values = re.findall(r'\b0[xX][0-9a-fA-F]+\b', content)
        return hex_values

filename = 'sample.txt'

hex_values = extract_hex_values(filename)
print("Hexadecimal values found:", hex_values)

# Extract All Capitalized Words

import re

def extract_capitalized_words(filename):
    with open(filename, 'r') as file:
        content = file.read()
        capitalized_words = re.findall(r'\b[A-Z][a-z]*\b', content)
        return capitalized_words

# Example usage
filename = 'sample.txt'

capitalized_words = extract_capitalized_words(filename)
print("Capitalized words found:", capitalized_words)
