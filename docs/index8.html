# Combine Get and Set for Dynamic Operations

from magicattr import get, set

class Values:
    def __init__(self, count):
        self.count = count

class Data:
    def __init__(self):
        self.values = Values(5)

data = Data()

current_count = get(data, "values.count")
set(data, "values.count", current_count * 2)

print(data.values.count)

10

Kwargs-Basics
Thu 05 December 2024 
 # reading data by using the kwargs

def greet(**kwargs):
    for key, value in kwargs.items():
        print(f"{key}: {value}")

def main_fun():
    greet(name="Alice", age=25, city="New York")

# Example usage
main_fun()

name: Alice
age: 25
city: New York
