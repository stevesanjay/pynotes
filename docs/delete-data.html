import pandas as pd

def delete_data_from_excel(file_name):
    df = pd.read_excel(file_name)

    dfs = df[df["name"]!= 'sanjay']

    dfs.to_excel(file_name,index=False)

    return dfs

delete_data_from_excel("data.xlsx")
