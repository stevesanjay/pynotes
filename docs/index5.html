# Extract Dates in DD/MM/YYYY Format

import re

def extract_dates(filename):
    with open(filename, 'r') as file:
        content = file.read()
        dates = re.findall(r'\b\d{2}/\d{2}/\d{4}\b', content)
        return dates

filename = 'sample.txt'

dates = extract_dates(filename)
