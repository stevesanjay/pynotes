# Pie Chart

# import necessary library

import matplotlib.pyplot as plt

# Data

sizes = [10, 20, 30, 40]

labels = ['A', 'B', 'C', 'D']

colors = ['gold', 'yellowgreen', 'lightcoral', 'lightskyblue']

# Creating the pie chart

plt.pie(sizes, labels=labels, colors=colors, autopct='%1.1f%%', startangle=140)
