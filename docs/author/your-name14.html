# Using Optional for Nullable Parameters

from typing import Optional

def greet(name: Optional[str] = None) -> str:
    if name:
        return f"Hello, {name}!"
    return "Hello, Guest!"

greet()

'Hello, Guest!'

greet("Alice")

'Hello, Alice!'
