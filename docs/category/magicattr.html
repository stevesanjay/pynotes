# Access Nested Attributes

from magicattr import get

class Address:
    def __init__(self, city):
        self.city = city

class Person:
    def __init__(self, name, address):
        self.name = name
        self.address = address

address = Address("coimbatore")

person  = Person("steve", address)

city  = get(person, "address.city")

city

'coimbatore'

from magicattr import get

class Calculator:
    def add(self, x, y):
        return x + y

calc = Calculator()

add_method = get(calc, "add")
result = add_method(5, 3)

print(result)

from magicattr import delete

class Profile:

        self.email = email

# Combine Get and Set for Dynamic Operations

from magicattr import get, set

class Values:
    def __init__(self, count):
        self.count = count

class Data:
    def __init__(self):
        self.values = Values(5)

data = Data()

current_count = get(data, "values.count")
set(data, "values.count", current_count * 2)

print(data.values.count)

# Get Nested Attributes in a Deeply Nested Object

from magicattr import get

class Inner:
    def __init__(self, value):
        self.value = value

class Outer:
    def __init__(self, inner):
        self.inner = inner

inner = Inner("Hello")

outer = Outer(inner)

# Access deeply nested attribute
value = get(outer, "inner.value")

value

'Hello'
