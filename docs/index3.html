# Count the Number of Words in a File

import re

def count_words(filename):
    with open(filename, 'r') as file:
        content = file.read()
        words = re.findall(r'\b\w+\b', content)
        return len(words)

# Example usage
filename = 'sample.txt'

word_count = count_words(filename)

print(f"Number of words: {word_count}")
